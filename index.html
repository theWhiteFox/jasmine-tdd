What is TDD?

  • Don't write a line of new code unless you first have a failing automated test.
  • Eliminate duplication.

What is TDD?

  • Red — write a short failing test
  • Green — make the test pass in the simplest way possible
  • Refactor — eliminate duplication introduced in the previous step

Why practice TDD?

  • Reduces number of defects
  • Allows refactoring with confidence
  • Leads to simple designs
  • Documents behavior of code

Jasmine has lots of useful matchers


describe('square', function () {
  it('squares the number', function () {
    expect(square(1)).toEqual(1);

    expect(square("a")).toBeUndefined;
  });
});

Top Contributers

Let's test drive a sample angular application that shows the list of top github contributers in Dublin.

Test setup


describe('UsersController', function () {
  var $scope;

  beforeEach(module('topContributors'));

  beforeEach(inject(function ($rootScope,
                              $controller) {
    $scope = $rootScope.$new();

    $controller('UsersController');
  }));

  // tests go here
});

First test


describe('UsersController', function () {
  // setup

  describe('when the controller starts', function () {
    it('loads the message', function () {
      expect($scope.message).toEqual('hello');
    });
  });
});

Setup for asynchronous test


describe('UsersController', function () {
  var $scope, githubService, githubDeferred;
  beforeEach(module('topContributors'));

  beforeEach(inject(function ($rootScope, $controller, $q,
                              _githubService_) {
    $scope = $rootScope.$new();
    githubService = _githubService_;

    githubDeferred = $q.defer();
    spyOn(githubService, 'fetchUsers')
            .and.returnValue(githubDeferred.promise);

    $controller('UsersController');
  }));
});

Asynchronous test


describe('UsersController', function () {
  // setup

  describe('when the controller starts', function () {
    it('fetches the users from GitHub', function () {
      githubDeferred.resolve([{login: 'testUser'}]);
      $scope.$apply();

      expect($scope.users).toEqual([{login: 'testUser'}]);
    });
  });
});
